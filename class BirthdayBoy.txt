class BirthdayBoy:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    def birthday(self):
        self.age += 1
boy_instance = BirthdayBoy(name="John", age=25)
print(f"{boy_instance.name}'s age before birthday: {boy_instance.age}")
boy_instance.birthday()
print(f"{boy_instance.name}'s age after birthday: {boy_instance.age}")
